    <style>
        body { font-family: Arial, sans-serif; text-align: center; margin: 20px; }
        input { width: 80%; padding: 10px; margin: 10px; }
        .cocktail { border: 1px solid #ddd; padding: 10px; margin: 10px; }
    </style>
</head>
<body>

    <h2>Cocktail Search</h2>
    <input type="text" id="search" placeholder="Search cocktails..." onkeyup="filterCocktails()">

    <div id="cocktail-list"></div>

    <script>
        let cocktails = [];

        async function fetchCocktailData() {
            const sheetID = "1abcXYZ12345";  // Replace with your actual Sheet ID
            const url = `https://docs.google.com/spreadsheets/d/${sheetID}/gviz/tq?tqx=out:json`;

            const response = await fetch(url);
            const text = await response.text();
            const json = JSON.parse(text.substring(47, text.length - 2)); // Cleaning Google's response

            cocktails = json.table.rows.map(row => ({
                name: row.c[0]?.v || "",
                season: row.c[1]?.v || "",
                ingredients: row.c[2]?.v || "",
                instructions: row.c[3]?.v || "",
                glass: row.c[4]?.v || "",
                garnish: row.c[5]?.v || ""
            }));

            displayCocktails(cocktails);
        }

        function displayCocktails(list) {
            const container = document.getElementById("cocktail-list");
            container.innerHTML = "";
            list.forEach(cocktail => {
                container.innerHTML += `
                    <div class="cocktail">
                        <h3>${cocktail.name}</h3>
                        <p><strong>Season:</strong> ${cocktail.season}</p>
                        <p><strong>Ingredients:</strong> ${cocktail.ingredients}</p>
                        <p><strong>Instructions:</strong> ${cocktail.instructions}</p>
                        <p><strong>Glass:</strong> ${cocktail.glass}</p>
                        <p><strong>Garnish:</strong> ${cocktail.garnish}</p>
                    </div>
                `;
            });
        }

        function filterCocktails() {
            const query = document.getElementById("search").value.toLowerCase();
            const filtered = cocktails.filter(cocktail =>
                cocktail.name.toLowerCase().includes(query) ||
                cocktail.ingredients.toLowerCase().includes(query) ||
                cocktail.season.toLowerCase().includes(query)
            );
            displayCocktails(filtered);
        }

        fetchCocktailData();
    </script>

</body>
</html>
